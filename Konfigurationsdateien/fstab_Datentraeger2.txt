# /var/log — sync-nah, kein noatime (Timestamps bei Logs sinnvoll)
LABEL=var-log  /var/log  xfs  defaults,nodev,nosuid,noexec  0 0

# /opt — lesedominant, noatime
LABEL=opt  /opt  xfs  defaults,nofail,nodev,noatime  0 0

# ~/.cache/ccache — max Performance, noatime, noexec
LABEL=ccache  /home/giant/.cache/ccache  xfs  defaults,nofail,nodev,nosuid,noexec,noatime  0 0

# ~/.cache/build — wie ccache + largeio für Buildartefakte
LABEL=build  /home/giant/.cache/build  xfs  defaults,nofail,nodev,nosuid,noexec,noatime,largeio  0 0

# /var/lib/libvirt — allocsize für große VM-Image-Allokation
LABEL=libvirt  /var/lib/libvirt  xfs  defaults,nofail,nodev,nosuid,allocsize=64m  0 0
